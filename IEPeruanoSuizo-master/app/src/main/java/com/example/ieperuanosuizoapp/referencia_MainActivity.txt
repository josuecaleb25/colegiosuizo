    private void setupTabLayout() {
        TabLayout.Tab tabAsistencia = tabLayoutLeaderboard.newTab().setText("Asistencia");
        TabLayout.Tab tabPuntualidad = tabLayoutLeaderboard.newTab().setText("Puntualidad");

        tabLayoutLeaderboard.addTab(tabAsistencia);
        tabLayoutLeaderboard.addTab(tabPuntualidad);

        tabLayoutLeaderboard.addOnTabSelectedListener(new TabLayout.OnTabSelectedListener() {
            @Override
            public void onTabSelected(TabLayout.Tab tab) {
                if (tab.getPosition() == 0) {
                    selectedMetric = "asistencia";
                } else {
                    selectedMetric = "puntualidad";
                }
                renderLeaderboard();
            }

            @Override
            public void onTabUnselected(TabLayout.Tab tab) {}

            @Override
            public void onTabReselected(TabLayout.Tab tab) {}
        });
    }

    private void setupMonthDropdown() {
        List<String> months = new ArrayList<>();
        months.add("Junio 2026");
        months.add("Mayo 2026");
        months.add("Abril 2026");

        ArrayAdapter<String> adapter = new ArrayAdapter<>(this, android.R.layout.simple_dropdown_item_1line, months);
        if (dropdownMonth != null) {
            dropdownMonth.setAdapter(adapter);
            dropdownMonth.setOnItemClickListener((parent, view, position, id) -> {
                selectedMonth = months.get(position);
                renderLeaderboard();
            });
        }

    private void setupRecyclerView() {

    private void renderLeaderboard() {
        Map<String, List<Player>> metricsMap = monthlyMetricData.get(selectedMonth);
        if (metricsMap == null) return;

        List<Player> originalList = metricsMap.get(selectedMetric);
        if (originalList == null) return;

        // Make deep copy of the mock records lists to avoid reference mutations
        List<Player> players = new ArrayList<>();
        for (Player p : originalList) {
            players.add(p.copy());
        }

        // Apply classroom filtering and sorting if toggle is active
        if (filterClassroomActive) {
            List<Player> filtered = new ArrayList<>();
            for (Player p : players) {
                if (activeUserAulaId.equals(p.getAulaId())) {
                    filtered.add(p);
                }
            }
            players = filtered;
            
            // Sort remaining filtered list descending by score to ensure correct order
            Collections.sort(players, (p1, p2) -> Integer.compare(p2.getScore(), p1.getScore()));
        }

        // Apply dynamic consecutive ranks of local display scope (1 to N)
        for (int i = 0; i < players.size(); i++) {
            players.get(i).setRank(i + 1);
        }

        // Bind Suffix based on metric selection
        String suffix = "asistencia".equals(selectedMetric) ? " días" : " días temprano";

        // Register suffix dynamically inside the adapter
        leaderboardAdapter.setMetricSuffix(suffix);

        // Bind Podium items safely matching available items size
        if (players.size() >= 1) {
            Player rank1 = players.get(0);
            tvPodiumName1.setText(rank1.getName());
            tvPodiumAvatarText1.setText(rank1.getAvatarInitials());
            tvPodiumScore1.setText(String.valueOf(rank1.getScore()));
            tvPodiumSuffix1.setText(suffix);
            tvPodiumTrendText1.setText(rank1.getTrendText());
            layoutPodiumTrendCapsule1.setVisibility(View.VISIBLE);
            setTrendPodiumCapsule(layoutPodiumTrendCapsule1, ivPodiumTrendIcon1, tvPodiumTrendText1, rank1.getTrendType());
        } else {
            tvPodiumName1.setText("Sem Datos");
            tvPodiumAvatarText1.setText("?");
            tvPodiumScore1.setText("");
            tvPodiumSuffix1.setText("");
            tvPodiumTrendText1.setText("");
            layoutPodiumTrendCapsule1.setVisibility(View.INVISIBLE);
        }

        if (players.size() >= 2) {
            Player rank2 = players.get(1);
            tvPodiumName2.setText(rank2.getName());
            tvPodiumAvatarText2.setText(rank2.getAvatarInitials());
            tvPodiumScore2.setText(String.valueOf(rank2.getScore()));
            tvPodiumSuffix2.setText(suffix);
            tvPodiumTrendText2.setText(rank2.getTrendText());
            layoutPodiumTrendCapsule2.setVisibility(View.VISIBLE);
            setTrendPodiumCapsule(layoutPodiumTrendCapsule2, ivPodiumTrendIcon2, tvPodiumTrendText2, rank2.getTrendType());
        } else {
            tvPodiumName2.setText("");
            tvPodiumAvatarText2.setText("");
            tvPodiumScore2.setText("");
            tvPodiumSuffix2.setText("");
            tvPodiumTrendText2.setText("");
            layoutPodiumTrendCapsule2.setVisibility(View.INVISIBLE);
        }

        if (players.size() >= 3) {
            Player rank3 = players.get(2);
            tvPodiumName3.setText(rank3.getName());
            tvPodiumAvatarText3.setText(rank3.getAvatarInitials());
            tvPodiumScore3.setText(String.valueOf(rank3.getScore()));
            tvPodiumSuffix3.setText(suffix);
            tvPodiumTrendText3.setText(rank3.getTrendText());
            layoutPodiumTrendCapsule3.setVisibility(View.VISIBLE);
            setTrendPodiumCapsule(layoutPodiumTrendCapsule3, ivPodiumTrendIcon3, tvPodiumTrendText3, rank3.getTrendType());
        } else {

            tvPodiumScore3.setText("");
            tvPodiumSuffix3.setText("");
            tvPodiumTrendText3.setText("");
            layoutPodiumTrendCapsule3.setVisibility(View.INVISIBLE);
        }

        // 3. Bind Remaining lists items to the adapter
        List<Player> remainingPlayers = new ArrayList<>();
        if (players.size() > 3) {
            remainingPlayers.addAll(players.subList(3, players.size()));
        }

        leaderboardAdapter.submitList(remainingPlayers);
        leaderboardAdapter.notifyDataSetChanged();
    }

    private void setTrendPodiumCapsule(View capsule, ImageView iv, TextView tv, int trendType) {
        if (trendType > 0) {
            if (iv != null) iv.setImageResource(R.drawable.ic_arrow_up_green);
            if (tv != null) {
                tv.setTextColor(Color.parseColor("#10B981")); // Emerald green
            }
            if (capsule != null) {
                capsule.setBackgroundResource(R.drawable.bg_trend_up_capsule);
            }
        } else if (trendType < 0) {
            if (iv != null) iv.setImageResource(R.drawable.ic_arrow_down_red);
            if (tv != null) {
                tv.setTextColor(Color.parseColor("#EF4444")); // Crimson red
            }
            if (capsule != null) {
                capsule.setBackgroundResource(R.drawable.bg_trend_down_capsule);
            }
        } else {
            if (iv != null) iv.setImageResource(R.drawable.ic_stable_gray);
            if (tv != null) {
                tv.setTextColor(Color.parseColor("#64748B")); // Slate secondary gray
            }
            if (capsule != null) {
                capsule.setBackgroundResource(R.drawable.bg_trend_stable_capsule);
            }
        }
    }
